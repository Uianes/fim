      program ImprimirNumeros;

      var
        i: integer;
        
      begin
        writeln('Programa que escreve dez números na tela:');
        
        // Inicialização da variável de controle
        i := 1;
      
        // Loop para imprimir os números de 1 a 10 usando WHILE
        while i <= 10 do
        begin
          writeln(i);
          // Incremento da variável de controle
          i := i + 1;
        end;

      end.
